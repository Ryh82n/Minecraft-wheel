  <h1>Minecraft Challenge Spinner</h1>
  <div id="pointer"></div>
  <div id="wheel"></div>
  <button id="spinBtn">Spin the Wheel</button>

  <!-- Sound -->
  <audio id="saadSound" src="saad-noob.mp3" preload="auto"></audio>

  <script>
    const challenges = [
      "Play with no armor",
      "Only use wooden tools",
      "Survive only on bread",
      "Don't jump for 10 minutes",
      "Play in F5 mode only",
      "Mine with a wooden pickaxe",
      "Play with inverted mouse",
      "Don't kill mobs",
      "Play with 2 hearts max",
      "No crafting allowed"
    ];

    const wheel = document.getElementById("wheel");
    const spinBtn = document.getElementById("spinBtn");
    const saadSound = document.getElementById("saadSound");

    let angle = 360 / challenges.length;

    // Build the wheel segments
    challenges.forEach((challenge, index) => {
      const segment = document.createElement("div");
      segment.className = "segment";
      segment.innerText = challenge;

      const rotation = angle * index;
      segment.style.transform = `rotate(${rotation}deg) translate(-100%)`;
      segment.style.background = index % 2 === 0 ? "#62c370" : "#4a8b61";

      wheel.appendChild(segment);
    });

    spinBtn.addEventListener("click", () => {
      const randomIndex = Math.floor(Math.random() * challenges.length);
      const rotateDeg = 360 * 5 + (360 - randomIndex * angle - angle / 2);

      wheel.style.transform = `rotate(${rotateDeg}deg)`;

      // Play sound after spin completes
      setTimeout(() => {
        saadSound.play();
        alert("Challenge: " + challenges[randomIndex]);
      }, 5000);
    });
  </script>
